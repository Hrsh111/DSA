// Topological Sorting/Ordering

// 1. Definition:
// - Topological sorting is a technique used for Directed Acyclic Graphs (DAGs).
// - It linearly orders vertices such that, for every directed edge u -> v, vertex u comes before v.

// 2. Valid Graph Type:
// - Only works for Directed Acyclic Graphs (DAGs).
// - Not applicable if the graph contains cycles.

// 3. Input Representation:
// - Adjacency list: Preferred for efficiency.
// - Adjacency matrix: Less common for large graphs.

// 4. Output:
// - A linear ordering of vertices satisfying dependency constraints.

//--------------------------------------------------------

// Algorithms for Topological Sorting:

// 1. Kahn's Algorithm (Using Indegree):
// - Iterative approach using a queue and indegree array.

// Steps:
// 1. Calculate indegree (number of incoming edges) for each node.
// 2. Add nodes with indegree 0 to a queue.
// 3. While the queue is not empty:
//    - Remove a node, add it to the result.
//    - Decrease indegree of its neighbors.
//    - Add neighbors with indegree 0 to the queue.
// 4. If all nodes are processed, the ordering is valid.

// Time Complexity: O(V + E)

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

void topologicalSort(int n, vector<vector<int>>& adj) {
    vector<int> indegree(n, 0);

    // Calculate indegree
    for (int i = 0; i < n; i++) {
        for (int neighbor : adj[i]) {
            indegree[neighbor]++;
        }
    }

    queue<int> q;
    for (int i = 0; i < n; i++) {
        if (indegree[i] == 0) q.push(i);
    }

    vector<int> topoOrder;
    while (!q.empty()) {
        int node = q.front(); q.pop();
        topoOrder.push_back(node);

        for (int neighbor : adj[node]) {
            indegree[neighbor]--;
            if (indegree[neighbor] == 0) q.push(neighbor);
        }
    }

    if (topoOrder.size() != n) {
        cout << "The graph is not a DAG (contains a cycle)." << endl;
        return;
    }

    for (int node : topoOrder) cout << node << " ";
    cout << endl;
}

int main() {
    int n = 6;
    vector<vector<int>> adj = {
        {2, 3}, {3, 4}, {3}, {}, {}, {}
    };

    topologicalSort(n, adj);
    return 0;
}

// 2. DFS-Based Algorithm:
// - Recursive approach using Depth-First Search.

// Steps:
// 1. Use a visited array to track nodes.
// 2. Perform DFS for each unvisited node.
// 3. Push each completed node to a stack.
// 4. Reverse the stack to get the ordering.

// Time Complexity: O(V + E)

#include <iostream>
#include <vector>
#include <stack>
using namespace std;

void dfs(int node, vector<vector<int>>& adj, vector<bool>& visited, stack<int>& st) {
    visited[node] = true;
    for (int neighbor : adj[node]) {
        if (!visited[neighbor]) dfs(neighbor, adj, visited, st);
    }
    st.push(node);
}

void topologicalSortDFS(int n, vector<vector<int>>& adj) {
    vector<bool> visited(n, false);
    stack<int> st;

    for (int i = 0; i < n; i++) {
        if (!visited[i]) dfs(i, adj, visited, st);
    }

    while (!st.empty()) {
        cout << st.top() << " ";
        st.pop();
    }
    cout << endl;
}

int main() {
    int n = 6;
    vector<vector<int>> adj = {
        {2, 3}, {3, 4}, {3}, {}, {}, {}
    };

    topologicalSortDFS(n, adj);
    return 0;
}
